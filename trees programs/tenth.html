bool hasDeadEnd(struct tree* root, int min, 
int max) {
    if (root == NULL)
        return false;
// Check if the current node is a dead end
    if (max - min <= 2) {
        return true;
    }
// Recursively check the left and 
right subtrees
    bool leftDeadEnd = hasDeadEnd(root->left, 
    min, root->data - 1);
    bool rightDeadEnd = hasDeadEnd(root->right, 
    root->data + 1, max);
// Return true if a dead end is found in 
either subtree
    return leftDeadEnd || rightDeadEnd;
}
