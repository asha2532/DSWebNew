#include<stdio.h>
            int isVowel(char ch)
            {
                ch=tolower(ch);
                return(ch=='a'|| ch=='e'|| ch=='i'|| ch=='o'|| ch=='u');
            }
            int countVowel(char str[])
            {
                int count=0;
                int i=0;
                printf("vowwel in the string:");
                while(str[i]!='\0')
                {
                    if(isVowel(str[i]))
                    {
                        count++;
                    }
                    i++;
                }
                return count;
            }
            int main()
            {
                char str[100];
