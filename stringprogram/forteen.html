#include<stdio.h>
            int stringToInteger(char *str)
            {
                int result=0;
                int sign=1;
                int i=0;
                if(str[0]=='-')
                {
                    sign=-1;
                    i=1;
                }
                while(str[i]!='\0')
                {
                    result=result*10+(str[i]-'0');
                    i++;
                }
                return sign*result;
            }
            int main()
            {
                char str[100];
                printf("enter a string:");
                scanf("%s",str);
                
                int intvalue=stringToInteger(str);
                printf("integer value:%d\n",intvalue);
                
                return 0;
            }
